To access your banking platform:

1. Get the API Gateway URL:
{{ if contains "NodePort" .Values.services.apiGateway.type }}
   export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "banking-platform.fullname" . }}-api-gateway)
   export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
   echo http://$NODE_IP:$NODE_PORT
{{ else if contains "LoadBalancer" .Values.services.apiGateway.type }}
   NOTE: It may take a few minutes for the LoadBalancer IP to be available.
         You can watch the status by running: kubectl get --namespace {{ .Release.Namespace }} svc -w {{ include "banking-platform.fullname" . }}-api-gateway
   export SERVICE_IP=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ include "banking-platform.fullname" . }}-api-gateway --template "{{"{{ range (index .status.loadBalancer.ingress 0) }}{{.}}{{ end }}"}}")
   echo http://$SERVICE_IP:{{ .Values.services.apiGateway.port }}
{{ else if contains "ClusterIP" .Values.services.apiGateway.type }}
   kubectl --namespace {{ .Release.Namespace }} port-forward svc/{{ include "banking-platform.fullname" . }}-api-gateway 8080:{{ .Values.services.apiGateway.port }}
   echo "Visit http://127.0.0.1:8080 to access the API Gateway"
{{ end }}

2. Get the Frontend URL:
{{ if contains "NodePort" .Values.services.frontend.type }}
   export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "banking-platform.fullname" . }}-frontend)
   export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
   echo http://$NODE_IP:$NODE_PORT
{{ else if contains "LoadBalancer" .Values.services.frontend.type }}
   export SERVICE_IP=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ include "banking-platform.fullname" . }}-frontend --template "{{"{{ range (index .status.loadBalancer.ingress 0) }}{{.}}{{ end }}"}}")
   echo http://$SERVICE_IP:{{ .Values.services.frontend.port }}
{{ else if contains "ClusterIP" .Values.services.frontend.type }}
   kubectl --namespace {{ .Release.Namespace }} port-forward svc/{{ include "banking-platform.fullname" . }}-frontend 3000:{{ .Values.services.frontend.port }}
